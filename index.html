// Modern KentScript Example
func fibonacci(n: int) -> list {
    if n <= 0 { return []; }
    if n == 1 { return [0]; }
    
    let sequence = [0, 1];
    while len(sequence) < n {
        let next = sequence[-1] + sequence[-2];
        sequence.push(next);
    }
    return sequence;
}

// Async HTTP request
async func fetchData(url: string) -> dict {
    import http;
    let response = await http.get(url);
    return json.parse(response.body);
}

// Multi-threading
let futures = [];
for url in urls {
    let future = thread fetchData(url);
    futures.push(future);
}

// Wait for all threads
let results = await Promise.all(futures);

Functions & Lambdas


// Function definition
func greet(name: string, times: int = 1) -> string {
    return "Hello " + name + "!".repeat(times);
}

// Lambda expressions
let square = (x: int) => x * x;
let add = (a: int, b: int) => a + b;

// Higher-order functions
func applyTwice(f: function, x: any) -> any {
    return f(f(x));
}

// Decorators
@logExecution
@validateParams
func process(data: list) -> list {
    return data.map(x => x * 2);
}

Database ORM

import db;

// Connect to database
let conn = db.connect("sqlite://data.db");

// Define models
model User {
    id: int @primary @auto;
    name: string;
    email: string @unique;
}

// Query with type safety
let users = conn.query(User)
    .where(u => u.age > 18)
    .orderBy("name")
    .limit(10);

ksc program.ks -o program.kbc

kvm program.kbc

# Debug mode
python kentscript.py --debug program.ks

# Profile performance
python kentscript.py --profile program.ks
